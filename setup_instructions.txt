This project uses a Next.js frontend, a Supabase PostgreSQL database, and a Python Random Forest machine learning model.

To run this simulation on a completely new laptop, follow these steps:

### 1. Prerequisites
- **Node.js**: Install Node.js (v18 or higher) from nodejs.org
- **Python 3**: Install Python (v3.9 or higher) from python.org

### 2. Node Dependencies
Open a terminal in the project folder and run:
```bash
npm install
```

### 3. Python Dependencies
The Random Forest ML model requires a few Python packages to run `model/predict_edges.py`. Run:
```bash
pip3 install scikit-learn numpy joblib
```
*(Depending on how Python is installed, you might use `pip` instead of `pip3`)*

### 4. Environment Variables
Create a file named `.env.local` in the root folder of the project containing your Supabase credentials:
```
NEXT_PUBLIC_SUPABASE_URL=your_supabase_url
NEXT_PUBLIC_SUPABASE_ANON_KEY=your_anon_key
------------------------------------------------------------------------------------------------------------------------------------
-- 1. game_rooms
CREATE TABLE game_rooms (
  id text PRIMARY KEY,
  status text NOT NULL,
  current_round int4 NOT NULL DEFAULT 1,
  total_rounds int4 NOT NULL DEFAULT 5,
  created_at timestamptz NOT NULL DEFAULT now(),
  current_origin text,
  current_destination text
);

-- 2. room_endpoints
CREATE TABLE room_endpoints (
  id int4 GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
  room_id text REFERENCES game_rooms(id) ON DELETE CASCADE,
  round int4,
  origin text,
  destination text
);

-- 3. traffic_edges
CREATE TABLE traffic_edges (
  id text PRIMARY KEY,
  from_node text NOT NULL,
  to_node text NOT NULL,
  free_time numeric NOT NULL,
  capacity int4 NOT NULL,
  base_flow int4 NOT NULL,
  current_flow int4 NOT NULL,
  travel_time numeric NOT NULL,
  created_at timestamptz NOT NULL DEFAULT now(),
  updated_at timestamptz NOT NULL DEFAULT now(),
  room_id text REFERENCES game_rooms(id) ON DELETE CASCADE
);

-- 4. simulation_sessions
CREATE TABLE simulation_sessions (
  id uuid PRIMARY KEY DEFAULT gen_random_uuid(),
  user_id text NOT NULL,
  current_round int4 NOT NULL DEFAULT 1,
  total_rounds int4 NOT NULL DEFAULT 5,
  grid_size int4 NOT NULL DEFAULT 5,
  is_complete bool NOT NULL DEFAULT false,
  current_origin text,
  current_destination text,
  created_at timestamptz NOT NULL DEFAULT now(),
  updated_at timestamptz NOT NULL DEFAULT now(),
  room_id text REFERENCES game_rooms(id) ON DELETE CASCADE,
  user_name text,
  has_submitted bool NOT NULL DEFAULT false
);

-- 5. round_logs
CREATE TABLE round_logs (
  id int8 GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
  session_id uuid REFERENCES simulation_sessions(id) ON DELETE CASCADE,
  round int4 NOT NULL,
  user_id text,
  origin text,
  destination text,
  chosen_route text,
  decision_latency numeric,
  predicted_time numeric,
  realized_time numeric,
  route_a_flow numeric,
  route_b_flow numeric,
  route_c_flow numeric,
  grid_size int4,
  route_path text[],
  route_edges jsonb,
  created_at timestamptz NOT NULL DEFAULT now(),
  choice_reason text,
  choice_reason_text text
);

-- 6. round_endpoints (old table, keep for compatibility)
CREATE TABLE round_endpoints (
  id int8 GENERATED ALWAYS AS IDENTITY PRIMARY KEY,
  session_id uuid REFERENCES simulation_sessions(id) ON DELETE CASCADE,
  round int4,
  origin text,
  destination text,
  created_at timestamptz NOT NULL DEFAULT now()
);

create policy "dev_all_game_rooms" on public.game_rooms for all using (true) with check (true);

create policy "dev_all_traffic_edges" on public.traffic_edges for all using (true) with check (true);
create policy "dev_all_round_endpoints" on public.round_endpoints for all using (true) with check (true);
create policy "dev_all_round_logs" on public.round_logs for all using (true) with check (true);
```

chmod +x start.sh
./start.sh





Open `http://localhost:3000` in your browser to start the simulation!
